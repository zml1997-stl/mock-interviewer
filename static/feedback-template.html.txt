    <script>
        document.addEventListener('DOMContentLoaded', function() {
            // Convert markdown-like formatting in feedback
            const feedbackContent = document.querySelector('.feedback-content');
            let html = feedbackContent.innerHTML;
            
            // Format headers
            html = html.replace(/^# (.+)$/gm, '<h3 class="mt-4 mb-3">$1</h3>');
            html = html.replace(/^## (.+)$/gm, '<h4 class="mt-3 mb-2">$1</h4>');
            html = html.replace(/^### (.+)$/gm, '<h5 class="mt-2 mb-2">$1</h5>');
            
            // Format lists
            html = html.replace(/^\d+\. (.+)$/gm, '<li>$1</li>');
            html = html.replace(/^- (.+)$/gm, '<li>$1</li>');
            html = html.replace(/(<li>.+<\/li>\n)+/g, function(match) {
                return '<ol>' + match.replace(/^- /gm, '') + '</ol>';
            });
            
            // Format paragraphs
            html = html.replace(/(.+)\n\n/g, '<p>$1</p>');
            
            // Format bold
            html = html.replace(/\*\*(.+?)\*\*/g, '<strong>$1</strong>');
            
            // Format italics
            html = html.replace(/\*(.+?)\*/g, '<em>$1</em>');
            
            feedbackContent.innerHTML = html;
            
            // Add proper list formatting
            const lists = document.querySelectorAll('.feedback-content ol');
            lists.forEach(list => {
                const items = list.querySelectorAll('li');
                items.forEach(item => {
                    // Check if it's a numbered list item
                    if (item.textContent.match(/^\d+\. /)) {
                        item.textContent = item.textContent.replace(/^\d+\. /, '');
                        list.setAttribute('type', '1');
                    } else {
                        // It's a bulleted list
                        item.textContent = item.textContent.replace(/^- /, '');
                        list.setAttribute('type', 'disc');
                    }
                });
            });
        });
    </script>
